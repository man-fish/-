# 组件继承

### 继承

```js
const CompVue = Vue.extend(component)

new CompVue({
    el:"#root",
    propsData:{
        propOne:"xxx"
        //子组件使用父组件的props
    },
    data:{
        text:0
        //子组件覆盖掉父组件的data
    }
})

const component2 = {
    extends : component,
    data (){
        return{
            text : 4
        }
    },
    mounted(){
        console.log("comp2 extends comp!")
        console.log(this.$parent.$options.name)
        this.$parent.text = 333
        //parent用来查看不要用来修改
    }
}

new Vue({
    el:'#root',
    components : {
        Comp2: component2
    } ,
    template:'<comp2></comp2>',
})
```
