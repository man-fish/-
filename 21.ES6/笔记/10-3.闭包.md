# 闭包

#### 1.闭包生成相同或者不同的随机数

```javascript
    +function(){
        function random(){
            var num = Math.floor(Math.random()*100);
            return function(){
                return num;
            }
        }
		
        var random_num1 = random();
        var num1 = random_num1();
        var num2 = random_num1();
        var random_num2 = random();
        console.log(num1,num2,random_num2());
		//😡num1 = num2 != random_num2();
    }();
```

#### 2.闭包生成缓存自增数

```javascript
    +function(){
        function addnum(){
            var num = 10
            //这个函数外层只会创建的时候调用一次，所以会形成一个数据缓存。
            return function(){
                num++
                console.log(num)
            }
        }
        var additional_num1 = addnum()
        var additional_num2 = addnum()
        setInterval(function(){
          additional_num1()
          additional_num2()
        },1000)
    }();
	|10	10|
	|11	11|	
	|12	12|
	|13	13|
	|14	14|
	|.	 .|
	|.	 .|
	|.	 .|
```
