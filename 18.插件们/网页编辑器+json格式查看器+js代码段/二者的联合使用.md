            $(function() {
            var editor = CodeMirror.fromTextArea(document.getElementById("json-input"), {
                mode: "javascript", //输入的语言模式
                lineNumbers: true, //是否显示每一行的行数
                readOnly: false, //只读
                styleActiveLine: true, //当前行高亮
                lineWrapping: true, // 自动换行
                matchBrackets: true //括号匹配
            });　　　　　
            editor.setOption("theme", "default"); //把内容设置为主题色，
                //--------------------------初始化编辑器
//这个是假数据，假设是从后台拿到的，需要展示到页面中
            var obj = {
                "message": "love",
                "code": 200,
                "response": {
                    "totalCount": 1,
                    "results": [{
                        "status": "deployed",
                        "login_port": 22,
                        "role_info": null,
                        "created": "2017-06-19 09:58:04",
                        "login_user": "root",
                        "hostname": "deploy-226",
                        "login_ip": "172.24.6.226",
                        "id": "1147edbde1494df696019fdb094be43d"
                    }],
                    "pageSize": 5,
                    "page": 1
                },
                "success": true
            }
				//填入数据到编辑器
                editor.setValue(JSON.stringify(obj, null, 4))
                
                //JSON.stringify() //方法的第三个参数的目的就是保留格式的，如果没有的话，
                //普通的JSON.stringify(obj)仅仅就是把对象转字符串，并不辉保留空格和换行
//实用的时候直接删掉中间的代码
                
                $('#btn-json-viewer').click(function() {
                    var json = editor.getValue();
                    //用户使用的时候才有用。
                    try {
                        var input = eval('(' + json + ')');
                    } catch (error) {
                        return alert("Cannot eval JSON: " + error);
                    }
                    var options = {
                        collapsed: $('#collapsed').is(':checked'),
                        withQuotes: $('#with-quotes').is(':checked')
                    };
                    $('#json-renderer').jsonViewer(input, options);
                });
                $('#btn-json-viewer').click()
            });
				数据联合展示
        </script>
